#### 设定固定IP

```yaml
su -
cd /etc/netplan
vi 50_cloud_init.yaml

network:
    ethernets:
        enp0s3:
            dhcp4: true
        enp0s8:
            addresses: [192.168.56.108/24]
            gateway4: 192.168.56.2
            nameservers:
                addresses: [192.168.56.2]
            # optional: true
    version: 2

netplan apply
ip a
```

#### 设定hostname
```
vi /etc/cloud/cloud.cfg
preserve_hostname: true

cat /etc/hostname
sudo hostnamectl set-hostname foobar

cat /etc/hosts
sudo reboot now
```
